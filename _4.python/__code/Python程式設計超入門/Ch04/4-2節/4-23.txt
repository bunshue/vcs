class animalBaseClass:

	def __init__(self, num):

		self.animallegs = num

	def walk(self):

		print('走動')

	def cry(self):

		print('吼叫')

	def getLegsNum(self):

		print(self.animallegs)


class snakeClass(animalBaseClass):

	def __init__(self, num):

		parent_class = super(snakeClass, self)

		parent_class.__init__(num)

		print('我是蛇')


nyoro = snakeClass(0)


nyoro.getLegsNum()
